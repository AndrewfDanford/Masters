\chapter{Reproducibility Checklist}

\section{Code and Environment}
\begin{itemize}
  \item Repository commit hash and branch.
  \item Dependency versions.
  \item Hardware and runtime configuration.
\end{itemize}

\section{Data Processing}
\begin{itemize}
  \item Inclusion and exclusion criteria.
  \item Split generation policy.
  \item Label mapping and uncertain label handling.
\end{itemize}

\section{Training and Evaluation}
\begin{itemize}
  \item Hyperparameters and early stopping criteria.
  \item Random seeds and number of runs.
  \item Full metric definitions and confidence interval method.
\end{itemize}
